% --- GLOSSARY ENTRIES ---
\newglossaryentry{neurodivergent}{
    name=neurodivergent,
    description={A term used to describe people whose brains function, learn, and process information differently than what is considered typical}
}

\newglossaryentry{executive_dysfunction}{
    name=executive dysfunction,
    description={A range of cognitive difficulties affecting the ability to plan, focus attention, and multi-task}
}

\newglossaryentry{hyperfocus}{
    name = hyperfocus,
    description = {A state of intense focus and concentration, often seen in individuals with ADHD, where they become deeply engrossed in a task or activity, sometimes to the exclusion of everything else}
}

\newglossaryentry{dyslexia}{
    name = dyslexia,
    description = {A learning disability that affects reading and writing skills, often characterised by difficulties with accurate and/or fluent word recognition and by poor spelling abilities}
}

\newglossaryentry{task_paralysis}{
    name = task paralysis,
    description = {A condition where an individual becomes unable to initiate or complete a task, often due to cognitive overload or anxiety}
}

\newglossaryentry{cognitive_tax}{
    name = cognitive tax,
    description = {The mental effort required to process and manage information, which can become overwhelming and impact performance}
}

\newglossaryentry{split_attention}{
    name={split-attention effect},
    description={A phenomenon occurring when a user must divide their attention between multiple physically separated but cognitively related sources of information, increasing extraneous load [cite: 162]}
}

\newglossaryentry{redundancy_principle}{
    name={redundancy principle},
    description={A learning principle suggesting that providing unnecessary or repetitive information can hinder processing by overloading working memory [cite: 162]}
}

\newglossaryentry{working_memory}{
    name={working memory},
    description={A cognitive system with limited capacity that is responsible for temporarily holding and processing information [cite: 123, 137, 172]}
}

\newglossaryentry{scaffolding}{
    name={scaffolding},
    description={Temporary support structures provided to a learner to help them complete tasks that would otherwise be beyond their unassisted reach [cite: 154, 248]}
}

\newglossaryentry{social_model}{
    name={social model of disability},
    description={A perspective arguing that individuals are disabled by environmental and societal barriers rather than by their own neurological or physical differences [cite: 189, 191]}
}

\newglossaryentry{medical_model}{
    name={medical model of disability},
    description={A traditional view that defines disability as a medical "deficit" to be corrected or cured within the individual [cite: 187, 188]}
}

\newglossaryentry{double_empathy}{
    name={double empathy problem},
    description={The theory that communication breakdowns between neurodivergent and neurotypical people are a two-way issue caused by differing processing styles [cite: 83, 110]}
}

\newglossaryentry{state_recovery_cost}{
    name={state recovery cost},
    description={The high mental effort and time required for a neurodivergent individual to regain focus and context after an interruption [cite: 87, 204]}
}

\newglossaryentry{halation}{
    name={halation},
    description={A visual distortion where text appears to glow or vibrate against high-contrast backgrounds, often causing sensory stress for autistic individuals [cite: 238, 243]}
}

\newglossaryentry{soft_high_contrast}{
    name={soft high contrast},
    description={A visual style using high-visibility colors against deep, non-pure-black backgrounds to provide clarity without triggering halation [cite: 240, 244]}
}

\newglossaryentry{task_initiation}{
    name={task initiation},
    description={The executive function ability to begin a project or task without undue delay or "paralysis" [cite: 58, 106, 252]}
}

\newglossaryentry{non_destructive}{
    name={non-destructive restoration},
    description={A system design that allows users to instantly revert all interface changes to prevent anxiety over lost configurations [cite: 272, 358]}
}
